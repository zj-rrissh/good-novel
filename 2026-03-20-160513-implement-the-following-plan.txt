│ 二、注解字段设计                                                             │
│                                                                              │
│ @Lock                                                                        │
│                                                                              │
│ @Target(ElementType.METHOD)                                                  │
│ @Retention(RetentionPolicy.RUNTIME)                                          │
│ public @interface Lock {                                                     │
│     String key();                                    // SpEL 表达式，如      │
│ "'balance:' + #userId"                                                       │
│     LockType type() default LockType.REENTRANT;      // REENTRANT | FAIR |   │
│ READ | WRITE                                                                 │
│     long waitTime() default -1;                       // -1 = 使用全局默认值 │
│  (200ms)                                                                     │
│     long leaseTime() default -1;                      // -1 = 使用全局默认值 │
│  (5s)                                                                        │
│     TimeUnit timeUnit() default TimeUnit.MILLISECONDS;                       │
│     String failMessage() default "";                  // 自定义失败提示      │
│ }                                                                            │
│                                                                              │
│ LockType                                                                     │
│                                                                              │
│ public enum LockType { REENTRANT, FAIR, READ, WRITE }                        │
│                                                                              │
│ @Idempotent                                                                  │
│                                                                              │
│ @Target(ElementType.METHOD)                                                  │
│ @Retention(RetentionPolicy.RUNTIME)                                          │
│ public @interface Idempotent {                                               │
│     IdempotentStrategy strategy();                    // TOKEN | FINGERPRINT │
│     String fingerprint() default "";                  // SpEL，仅            │
│ FINGERPRINT 策略使用                                                         │
│     String tokenParam() default "idempotencyKey";     // 方法参数名，仅      │
│ TOKEN 策略使用                                                               │
│     long ttl() default 10;                            // 幂等窗口时长        │
│     TimeUnit timeUnit() default TimeUnit.MINUTES;                            │
│     boolean optional() default false;                 // TOKEN 模式下，true  │
│ = key 缺失时跳过而非报错                                                     │
│     String message() default "";                      // 自定义重复提交提示  │
│ }                                                                            │
│                                                                              │
│ IdempotentStrategy                                                           │
│                                                                              │
│ public enum IdempotentStrategy { TOKEN, FINGERPRINT }                        │

│ 三、AOP 切面核心处理流程                                                     │
│                                                                              │
│ 3.1 SpelExpressionEvaluator（共享工具）                                      │
│                                                                              │
│ evaluate(expression, method, args):                                          │
│   1. DefaultParameterNameDiscoverer 获取方法参数名                           │
│   2. StandardEvaluationContext 注册 参数名 -> 参数值                         │
│   3. 自动注入 #currentUserId（从 SecurityContext/CurrentUserHolder 获取）    │
│   4. SpelExpressionParser 解析并求值，返回 String                            │
│                                                                              │
│ 3.2 LockAspect（@Order(1) — 最外层）                                         │
│                                                                              │
│ @Around("@annotation(lock)")                                                 │
│ 1. SpEL 解析 lock.key() → resolvedKey                                        │
│ 2. 拼接完整 Redis key: "{keyPrefix}:lock:{resolvedKey}"                      │
│ 3. 解析 waitTime/leaseTime（注解值 >= 0 则用注解值，否则用                   │
│ RedissonProperties 全局默认值）                                              │
│ 4. 根据 lock.type() 从 RedissonClient 获取对应锁：                           │
│      REENTRANT → getLock()                                                   │
│      FAIR      → getFairLock()                                               │
│      READ      → getReadWriteLock().readLock()                               │
│      WRITE     → getReadWriteLock().writeLock()                              │
│ 5. rLock.tryLock(waitTime, leaseTime, MILLISECONDS)                          │
│ 6. 获取失败 → throw BusinessException(LOCK_ACQUIRE_FAILED, failMessage)      │
│ 7. 获取成功 → try { joinPoint.proceed() } finally { unlock if held }         │
│                                                                              │
│ 3.3 IdempotentAspect（@Order(2) — 锁内层）                                   │
│                                                                              │
│ @Around("@annotation(idempotent)")                                           │
│ 1. 判断策略：                                                                │
│    TOKEN:                                                                    │
│      - 从方法参数中按 tokenParam 名称提取 token 值                           │
│      - token 为空：optional=true 则跳过，optional=false 则抛                 │
│ IDEMPOTENCY_KEY_REQUIRED                                                     │
│    FINGERPRINT:                                                              │
│      - SpEL 解析 fingerprint 表达式 → rawFingerprint                         │
│      - SHA-256 哈希 → idempotencyKey                                         │
│ 2. 构建 IdempotencyScope(userId, requestPath, idempotencyKey)                │
│ 3. 调用 IdempotencyService.record(scope, ttl)                                │
│      返回 true → 首次请求，proceed()                                         │
│      返回 false → 重复请求，throw BusinessException(IDEMPOTENT_CONFLICT)     │
│                                                                              │
│ @Order 设计理由: Lock(@Order(1)) 包裹 Idempotent(@Order(2))。先获取锁再检查  │
│ 幂等，消除两个并发请求同时通过幂等检查的竞态窗口。                           │
│                                                                              │
